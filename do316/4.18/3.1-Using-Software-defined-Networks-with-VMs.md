```bash
# Create two web servers VM using the webservers namespace
virtctl create vm \
  --name=web1 \
  --instancetype=u1.small \
  --preference=rhel.10 \
  --volume-datasource=src:openshift-virtualization-os-images/rhel10-helloworld \
  --namespace webservers \
  --cloud-init-user-data "$(cat cloud-config.b64)"

virtctl create vm \
  --name=web2 \
  --instancetype=u1.small \
  --preference=rhel.10 \
  --volume-datasource=src:openshift-virtualization-os-images/rhel10-helloworld \
  --namespace webservers \
  --cloud-init-user-data "$(cat cloud-config.b64)"
```

Label the VMs

```bash
oc edit vm web1
```

<pre>
apiVersion: kubevirt.io/v1
kind: VirtualMachine
metadata:
  name: web1
spec:
...
  template:
    metadata:
      <b>labels:
        tier: web  # This label is the "link" to the service</b>
    spec:
      domain:
        devices:
          interfaces:
            - name: default
              masquerade: {} # Required for service connectivity
...
</pre>


Add a service that expose the Web VMs

<pre>
apiVersion: v1
kind: Service
metadata:
  name: webservers-svc
spec:
  type: ClusterIP
  <b>selector:
    tier: web</b>  # This MUST match the label in your VM template
  ports:
    - name: http
      protocol: TCP
      port: 80         # Port accessible to other pods
      targetPort: 80   # Port running inside the VM
</pre>


From another VM, check the access to the VM through the service.

```bash
curl webservers-svc.webservers.svc.cluster.local
# webservers-svc is service name
# webservers is the project name
```
